const axios = require('axios'); export default async function handler(req, res) { // 设置 CORS 头（在 vercel.json 中已经配置，这里作为备用） res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'GET, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); // 处理预检请求 if (req.method === 'OPTIONS') { res.status(200).end(); return; } // 只允许 GET 请求 if (req.method !== 'GET') { res.status(405).json({ error: 'Method not allowed' }); return; } try { // 获取查询参数 const { currency = 'usd', format = 'json' } = req.query; // 验证货币参数 const validCurrencies = ['usd', 'eur', 'gbp', 'jpy', 'cny', 'aud', 'cad', 'chf']; if (!validCurrencies.includes(currency.toLowerCase())) { res.status(400).json({ error: 'Invalid currency', valid_currencies: validCurrencies }); return; } // 调用 CoinGecko API const response = await axios.get('https://api.coingecko.com/api/v3/simple/price', { params: { ids: 'bitcoin', vs_currencies: currency, include_market_cap: 'true', include_24hr_vol: 'true', include_24hr_change: 'true', include_last_updated_at: 'true' }, headers: { 'User-Agent': 'Bitcoin-Price-API/1.0' }, timeout: 10000 }); const bitcoinData = response.data.bitcoin; if (!bitcoinData) { res.status(404).json({ error: 'Bitcoin data not found' }); return; } // 构造响应数据 const priceData = { success: true, timestamp: new Date().toISOString(), data: { price: bitcoinData[currency], market_cap: bitcoinData[`${currency}_market_cap`], volume_24h: bitcoinData[`${currency}_24h_vol`], price_change_24h: bitcoinData[`${currency}_24h_change`], last_updated: bitcoinData.last_updated_at ? new Date(bitcoinData.last_updated_at * 1000).toISOString() : null }, currency: currency.toUpperCase(), source: 'CoinGecko API' }; // 根据 format 参数返回不同格式 if (format === 'text') { res.setHeader('Content-Type', 'text/plain'); res.status(200).send(`Bitcoin Price: $${priceData.data.price.toLocaleString()} ${currency.toUpperCase()}`); } else { res.status(200).json(priceData); } } catch (error) { console.error('API Error:', error.message); // 处理不同的错误类型 if (error.response) { // API 响应错误 res.status(error.response.status || 500).json({ error: 'API Error', message: error.response.data?.error || 'Failed to fetch data from CoinGecko', status: error.response.status }); } else if (error.request) { // 网络错误 res.status(503).json({ error: 'Network Error', message: 'Unable to connect to CoinGecko API' }); } else { // 其他错误 res.status(500).json({ error: 'Internal Server Error', message: error.message }); } } }
